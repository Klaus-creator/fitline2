function submitBooking() {

// Eingabefelder auslesen

let selectedTermin = document.querySelector('input[name="termin"]:checked');

let name = document.getElementById("name").value.trim();

let email = document.getElementById("email").value.trim();

let teampartner = document.getElementById("teampartner").value; // Teampartnernummer aus dem versteckten Feld



// Überprüfen, ob alle Felder ausgefüllt sind

if (!selectedTermin || name === "" || email === "") {

console.error("Bitte wählen Sie einen Termin und geben Sie Name und E-Mail ein.");

return;

}



// Überprüfen, ob die Teampartnernummer vorhanden ist

if (!teampartner) {

console.error("Fehler: Kein Teampartner gefunden");

return; // Verhindere das Absenden des Formulars

}



// Formulardaten vorbereiten

let formData = {

termin: selectedTermin.value,

name: name,

email: email,

teampartner: teampartner, // Teampartner in das Formular einfügen

teilgenommen: 'nein' // Standardwert für Teilnahme

};



console.log("Formulardaten:", formData); // Debugging-Log



// Formulardaten an den Make Webhook senden

fetch("https://hook.eu2.make.com/syx2i44241dc2ari4bn2g9gc31ee3gju", { // Ersetze mit deiner Make Webhook-URL

method: "POST",

body: JSON.stringify(formData),

headers: { "Content-Type": "application/json" }

})

.then(response => response.text())

.then(data => {

console.log("Antwort:", data); // Antwort ausgeben

document.getElementById("bookingForm").reset(); // Formular zurücksetzen

})

.catch(error => {

console.error("Fehler beim Absenden:", error);

});

}

</script>
